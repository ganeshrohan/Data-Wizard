# Netflix Data Concept 
# Databse used: Snowflake

This analytics model uses entities to represent physical tables with raw data, 
while metrics not actual columsn but are reusable calculations that make the data analysis-ready. 
Relationships are the defined logical rules that join these tables together. 

metrics:
  # --- Numeric Metrics ---
  - metric_name: "Invoice Amount (USD)"
    aliases: ["Revenue in Dollars", "USD Payments"]
    description: "The numeric value of an invoice ONLY IF it is in USD. Returns NULL for other currencies. Use this metric directly in a SUM() aggregation to get the total dollar revenue. To use this in SQL, you MUST use its 'calculation_logic' expression. This is NOT a physical column."
    applies_to_entity: "invoice"
    source_column: "total"
    calculation_logic: "CASE WHEN total LIKE '$%' THEN CAST(REGEXP_REPLACE(total, '[^0-9.]', '') AS DECIMAL(10,2)) ELSE NULL END"
    data_type: "DECIMAL(10,2)"
    nlq_examples:
      - natural_language_query: "What is the total revenue in dollars?"
        sql_snippet: "SELECT SUM(CASE WHEN i.total LIKE '$%' THEN CAST(REGEXP_REPLACE(i.total, '[^0-9.]', '') AS DECIMAL(10,2)) ELSE NULL END) FROM invoice i"
  - metric_name: "Invoice Amount (EUR)"
    description: "The numeric value of an invoice ONLY IF it is in EUR. Returns NULL for other currencies. Use this metric directly in a SUM() aggregation to get the total euro revenue. To use this in SQL, you MUST use its 'calculation_logic' expression. This is NOT a physical column."
    applies_to_entity: "invoice"
    source_column: "total"
    calculation_logic: "CASE WHEN total LIKE '€%' THEN CAST(REGEXP_REPLACE(total, '[^0-9.]', '') AS DECIMAL(10,2)) ELSE NULL END"

    description: "The original CURRENCY CODE ('USD', 'EUR') of the invoice. Use this for filtering or grouping by currency type. To use this in SQL, you MUST use its 'calculation_logic' expression."
    applies_to_entity: "invoice"

  # --- Date/Time Metrics ---
  - metric_name: "Typed Service Start Date"
    description: "The proper DATE type for the start of an invoice service period. Use this for all date-based filtering and operations on the service start."
    applies_to_entity: "invoice"
    source_column: "service_period"
    calculation_logic: "TO_DATE(SPLIT_PART(service_period, ' to ', 1), 'YYYY-MM-DD')"
    data_type: "DATE"
  - metric_name: "Typed Service End Date"
    description: "The proper DATE type for the end of an invoice service period. Use this for all date-based filtering and operations on the service end."
    applies_to_entity: "invoice"
    source_column: "service_period"
    calculation_logic: "TO_DATE(SPLIT_PART(service_period, ' to ', 2), 'YYYY-MM-DD')"
    data_type: "DATE"
  - metric_name: "Typed Stream Date"
    description: "The 'stream_date' column converted to a proper DATE type. Use this for all date-based operations."
    applies_to_entity: "streams"
    source_column: "stream_date"
    format_string: "DD-Mon-YYYY"
    calculation_logic: "TO_DATE(stream_date)"
    data_type: "DATE"
  - metric_name: "Year of Stream"
    description: "The year component of the stream date. Useful for annual reporting."
    applies_to_entity: "streams"
    source_column: "stream_date"
    calculation_logic: "EXTRACT(YEAR FROM TO_DATE(stream_date))"
    data_type: "INT"
  - metric_name: "Month of Stream"
    description: "The month number (1-12) of the stream date. Useful for monthly reporting."
    applies_to_entity: "streams"
    source_column: "stream_date"

    data_type: "INT"
  - metric_name: "Typed Invoice Date"

relationships:
  - name: "streams_to_customer"
    semantic_description: "Links a customer's individual streaming events to their main account profile."
    from_entity: "streams"
    to_entity: "customer"
    cardinality: "MANY_TO_ONE"
    join_on:
      - { from_column: "customer_id", to_column: "customer_id" }
  - name: "streams_to_content"
    semantic_description: "Connects a specific streaming event to the content that was watched."
    from_entity: "streams"
    to_entity: "content"
    cardinality: "MANY_TO_ONE"
    join_on:
      - { from_column: "content_id", to_column: "content_id" }
  - name: "invoices_to_customer"
    semantic_description: "Associates a customer's billing records with their main account profile."
    usage_hint: "Use this join for any query combining billing information (plan, total) with customer attributes (state, country)."
    from_entity: "invoice"
    to_entity: "customer"
    cardinality: "MANY_TO_ONE"
    join_on:
      - { from_column: "customer_id", to_column: "customer_id" }
  - name: "invoice_to_streams"
    semantic_description: "A multi-hop relationship connecting billing to streaming activity via the customer. Essential for analyzing viewing behavior of customers on specific plans (e.g., 'What do Premium users watch?')."
    from_entity: "invoice"
    to_entity: "streams"
    cardinality: "ONE_TO_MANY"
    join_path:
      - { from: "invoice", to: "customer", on: "customer_id" }
      - { from: "customer", to: "streams", on: "customer_id" }
